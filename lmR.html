In[25]:
plot(df$x, df$y,
     xlab = "", ylab = "",
     main = "Scatter Plot with Linear Regression Line")
lines(df$x, y_predict)
xlabel <- "x-axis"
ylabel <- "y-axis"
title(xlab = xlabel, col.lab = "pink")
title(ylab = ylabel, col.lab = "purple")

abline(LMR, col = "blue")
slope_annotation <- sprintf("Slope: %.2f", LMR$coefficients[2])
text(x = max(df$x) * 0.9, y = max(df$y) * 0.9, label = slope_annotation, col = "red")
R2_annotation <- sprintf("R^2: %.2f", summary(LMR)$r.squared)
text(x = max(df$x) * 0.9, y = max(df$y) * 0.85, label = R2_annotation, col = "red", adj = 0)
